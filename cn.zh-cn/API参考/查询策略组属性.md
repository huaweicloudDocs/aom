```
{
     "errorCode": SVCSTG.PE.0,
      "errorMessage": "",
      "config": {
          "id":"f9c7f57e-b1dc-4ef0-a009-ff2848776803",
          "max_instances": 100,//最大实例数，表示扩容最大限制
          "min_instances": 1,//最小实例数，表示扩容最小限制
          "cooldown_time": 60//冷却时间，应用的策略执行完后必须冷却该时间后才能继续执行下条策略
      }
}
```
